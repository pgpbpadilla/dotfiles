* Dotfiles

  User configuration files for various programs.

  For program-specific instructions look at the README files:

  #+begin_src bash :exports both
    for d in $(find . -not -name ".*" -type d -maxdepth 1 | sort);
    do
        readme="${d}/README.org"
        if [ ! -f ${readme} ]; then
           echo "${readme} [MISSING]"
           continue
        fi

       echo "[[${readme}][${readme}]]"
    done
  #+end_src
